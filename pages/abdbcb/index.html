如何遍历一个dom tree 

function travelsal(node) {
    if(node && node.nodeType === 1) { //是元素节点
        console.log(node.tagName)
    }
    var i = 0,childNodes = node.childNodes,item;
    for(;i < childNodes.length; i++) {
        item = childNodes[i];
        if(item.nodeType === 1) {
            travelsal(item)
        }
    }
}
